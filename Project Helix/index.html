Each group of 4 RNA bases = 8 bits = 1 ASCII character. Using the mapping A=00, U=01, G=10, C=11 (counting from 0 as the LOG_088 hint says):

import itertools

seq = "[ click to reveal — spoiler ]"
bases = ['A','U','G','C']

# Try all 24 permutations of bit assignments
for perm in itertools.permutations(['00','01','10','11']):
    mapping = dict(zip(bases, perm))
    result = ''
    for i in range(0, len(seq), 4):
        bits = ''.join(mapping[b] for b in seq[i:i+4])
        result += chr(int(bits, 2))
    if result.isprintable():
        print(f"Mapping: {mapping}")
        print(f"Result:  {result}")
